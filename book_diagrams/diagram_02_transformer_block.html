Both sub-layers follow the pattern: LayerNorm → operation → Add residual (Pre-LN variant, used in LLaMA/GPT-2). The residual connections (dashed blue lines) let gradients flow directly to earlier layers, enabling training of very deep networks. Output shape is identical to input shape, allowing arbitrary stacking.
Python Demo — stdlib only
import math, random

def layer_norm(x, eps=1e-5):
    "Normalize a vector to mean=0, std=1"
    mean = sum(x) / len(x)
    var = sum((v - mean)**2 for v in x) / len(x)
    return [(v - mean) / math.sqrt(var + eps) for v in x]

def gelu(x):
    "Gaussian Error Linear Unit activation"
    return x * 0.5 * (1.0 + math.tanh(math.sqrt(2/math.pi)*(x+0.044715*x**3)))

def linear(x, W):
    "Matrix-vector product: W is list of rows"
    return [sum(w*xi for w, xi in zip(row, x)) for row in W]

def ffn(x, W1, W2):
    hidden = [gelu(v) for v in linear(x, W1)]
    return linear(hidden, W2)

# Simulate one transformer sub-layer: LN → FFN → Add residual
d = 8   # d_model (try 16 or 32)
x = [random.gauss(0, 1) for _ in range(d)]
W1 = [[random.gauss(0, 0.1) for _ in range(d)] for _ in range(d*4)]
W2 = [[random.gauss(0, 0.1) for _ in range(d*4)] for _ in range(d)]

x_norm = layer_norm(x)
ffn_out = ffn(x_norm, W1, W2)
x_prime = [xi + fi for xi, fi in zip(x, ffn_out)]  # residual add

print(f"Input norm:  {[round(v,3) for v in x_norm]}")
print(f"FFN output:  {[round(v,3) for v in ffn_out]}")
print(f"After add:   {[round(v,3) for v in x_prime]}")
print(f"Shape preserved: input dim={len(x)} → output dim={len(x_prime)}")
# Try changing d=8 to d=16 to see the FFN expand to d*4=64 hidden units.
