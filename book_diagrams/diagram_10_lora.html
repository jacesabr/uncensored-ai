LoRA (Hu et al., 2021) hypothesizes that weight updates during fine-tuning have low intrinsic rank. By constraining ΔW = B·A where r ≪ d, the model trains only r·(d_in + d_out) parameters per layer. A initialized from Gaussian, B initialized to zero — so ΔW = 0 at start and training begins from the pretrained model. QLoRA (Dettmers et al., 2023) extends this to 4-bit quantized base models, enabling 65B fine-tuning on a single 48GB GPU.
Python Demo — stdlib only
import random, math

def matmul(A, B):
    "A: m×k, B: k×n → m×n"
    m, k, n = len(A), len(B), len(B[0])
    return [[sum(A[i][p]*B[p][j] for p in range(k)) for j in range(n)] for i in range(m)]

def matvec(M, x): return [sum(M[i][j]*x[j] for j in range(len(x))) for i in range(len(M))]

d, r = 8, 2   # try d=16, r=1 or r=4

# Pretrained weight W (frozen) — random for demo
W = [[random.gauss(0, 0.1) for _ in range(d)] for _ in range(d)]

# LoRA matrices: A ~ N(0,σ), B = 0 initially
A = [[random.gauss(0, 0.02) for _ in range(d)] for _ in range(r)]  # r×d
B = [[0.0 for _ in range(r)] for _ in range(d)]                    # d×r

x = [random.gauss(0, 1) for _ in range(d)]

# Full forward pass: h = W·x + B·A·x
Wx = matvec(W, x)
BAx = matvec(B, matvec(A, x))  # low-rank path
h = [a + b for a, b in zip(Wx, BAx)]

full_params = d * d
lora_params = r * d + d * r
print(f"d={d}, r={r} (rank)")
print(f"Full W params:   {full_params:,}")
print(f"LoRA A+B params: {lora_params:,}  ({lora_params/full_params*100:.1f}% of full)")
print(f"\nW·x output[:4]:     {[round(v,4) for v in Wx[:4]]}")
print(f"B·A·x (ΔW) [:4]:   {[round(v,4) for v in BAx[:4]]}  (≈ 0 at init)")
print(f"h = W·x + ΔW [:4]: {[round(v,4) for v in h[:4]]}")
