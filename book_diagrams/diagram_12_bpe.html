BPE starts with a character-level vocabulary and iteratively merges the most frequent adjacent pair. After k merges, the vocabulary has 256 (bytes) + k entries. The tokenizer is trained on the same corpus as the model — frequent subwords (like "-ing", "un-", "tion") become single tokens. This balances sequence length vs vocabulary size: rare words split into known subwords, avoiding the unknown-token problem of word-level tokenizers.
Python Demo — stdlib only
import collections, re

def get_pairs(vocab):
    """Count all adjacent pair frequencies across corpus."""
    pairs = collections.Counter()
    for word, freq in vocab.items():
        symbols = word.split()
        for i in range(len(symbols)-1):
            pairs[(symbols[i], symbols[i+1])] += freq
    return pairs

def merge_vocab(vocab, pair):
    """Replace all occurrences of pair in vocab."""
    merged = " ".join(pair)
    replacement = "".join(pair)
    new_vocab = {}
    for word, freq in vocab.items():
        new_vocab[word.replace(merged, replacement)] = freq
    return new_vocab

# Initial word-level vocab with character split + end marker
corpus = ["low", "low", "low", "lower", "lower", "lowest", "newest", "newer"]
vocab = collections.Counter()
for word in corpus:
    vocab[" ".join(list(word) + ["</w>"])] += 1

print("Initial vocab:", dict(vocab))
num_merges = 6  # try 3 or 10
merge_rules = []

for i in range(num_merges):
    pairs = get_pairs(vocab)
    if not pairs: break
    best = pairs.most_common(1)[0][0]
    vocab = merge_vocab(vocab, best)
    merge_rules.append(best)
    print(f"Merge {i+1}: {best[0]!r} + {best[1]!r} → {''.join(best)!r}  (count={pairs[best]})")

print(f"\nFinal vocab after {num_merges} merges:", list(vocab.keys()))
# Try num_merges=10 to see words fully merged into whole-word tokens.
