Full self-attention formula
Attention(Q, K, V) = softmax( QKᵀ / √d_k ) · V
← scale by √d_k to prevent gradient vanishing in high-dim dot products
Each token queries all others via its Q vector. High Q·K dot product = strong attention = large weight on that V. Scaling by √d_k stabilizes gradients when d_k is large. The output for each position is a weighted blend of all value vectors — this is how context flows across the sequence.
Python Demo — stdlib only
import math, random

def dot(a, b): return sum(x*y for x, y in zip(a, b))

def softmax(scores):
    m = max(scores)
    e = [math.exp(s - m) for s in scores]
    t = sum(e)
    return [v/t for v in e]

def attention(Q, K, V, d_k):
    """Single-head scaled dot-product attention."""
    scale = math.sqrt(d_k)
    outputs = []
    for q in Q:                              # each query token
        scores = [dot(q, k) / scale for k in K]  # dot with all keys
        weights = softmax(scores)
        out = [sum(w * v[i] for w, v in zip(weights, V))
               for i in range(len(V[0]))]
        outputs.append(out)
    return outputs, weights  # return last weights for display

d_k = 4   # key/query dimension  (try 8 or 16)
seq_len = 4

Q = [[random.gauss(0,1) for _ in range(d_k)] for _ in range(seq_len)]
K = [[random.gauss(0,1) for _ in range(d_k)] for _ in range(seq_len)]
V = [[random.gauss(0,1) for _ in range(d_k)] for _ in range(seq_len)]

outputs, last_weights = attention(Q, K, V, d_k)
tokens = ["cat", "sat", "on", "mat"]
print("Attention weights for last query:")
for tok, w in zip(tokens, last_weights):
    bar = "█" * int(w * 40)
    print(f"  {tok:5s} {bar} {w:.3f}")
