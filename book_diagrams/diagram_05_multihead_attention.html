Why multiple heads?
Each head can learn a different attention pattern: syntax, coreference, positional proximity. The model attends to multiple "aspects" simultaneously.
Computational cost
Each head uses d_k = d_model/h, so total FLOPs ≈ single-head attention. Parallelism is free — heads run simultaneously on GPU.
GPT-3 scale
d_model = 12,288, h = 96 heads, d_k = 128. Each head attends in 128-dim subspace, output projection is 12k×12k matrix.
Multi-head attention runs h attention functions in parallel, each in a d_k = d_model/h dimensional subspace. The concatenated outputs are linearly projected back to d_model. This allows the model to jointly attend to information from different representation subspaces at different positions.
Python Demo — stdlib only
import math, random

def dot(a, b): return sum(x*y for x,y in zip(a,b))
def softmax(v):
    m=max(v); e=[math.exp(x-m) for x in v]; t=sum(e)
    return [x/t for x in e]
def matmul(A, x): return [dot(row, x) for row in A]
def rand_mat(r, c): return [[random.gauss(0,0.1) for _ in range(c)] for _ in range(r)]

def single_head_attention(x_seq, Wq, Wk, Wv, d_k):
    Q = [matmul(Wq, x) for x in x_seq]
    K = [matmul(Wk, x) for x in x_seq]
    V = [matmul(Wv, x) for x in x_seq]
    scale = math.sqrt(d_k)
    outputs = []
    for q in Q:
        scores = softmax([dot(q, k)/scale for k in K])
        out = [sum(scores[j]*V[j][i] for j in range(len(V))) for i in range(d_k)]
        outputs.append(out)
    return outputs  # [seq_len × d_k]

d_model, h, seq_len = 8, 2, 3  # try h=4
d_k = d_model // h

x_seq = [[random.gauss(0, 1) for _ in range(d_model)] for _ in range(seq_len)]

head_outputs = []
for i in range(h):
    Wq, Wk, Wv = rand_mat(d_k, d_model), rand_mat(d_k, d_model), rand_mat(d_k, d_model)
    out = single_head_attention(x_seq, Wq, Wk, Wv, d_k)
    head_outputs.append(out)
    print(f"Head {i+1}: output[0][:3] = {[round(v,3) for v in out[0][:3]]}")

# Concatenate heads: [seq × (h*d_k)] = [seq × d_model]
concat = [[v for h_out in [head_outputs[hi][pos] for hi in range(h)] for v in h_out]
          for pos in range(seq_len)]
Wo = rand_mat(d_model, d_model)
final = [matmul(Wo, pos) for pos in concat]
print(f"\nMHA final output[0] dim: {len(final[0])} (== d_model={d_model})")
