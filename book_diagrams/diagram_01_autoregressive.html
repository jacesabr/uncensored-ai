At each step t, the model sees all previously generated tokens and produces a probability distribution over the full vocabulary (~32k–128k tokens). One token is sampled (or argmax-selected at temperature=0), appended to the sequence, and the process repeats. This is O(n²) attention cost per sequence.
Python Demo — stdlib only
import random, math

# Tiny vocabulary and bigram "model"
vocab = ["The", "cat", "sat", "on", "the", "mat", "and", ".", "<EOS>"]
bigram_probs = {
    "The": {"cat":0.8, "mat":0.2},
    "cat": {"sat":0.9, ".":0.1},
    "sat": {"on":1.0},
    "on":  {"the":1.0},
    "the": {"mat":0.7, "cat":0.3},
    "mat": {"and":0.5, ".":0.3, "<EOS>":0.2},
    "and": {"the":1.0},
    ".":   {"<EOS>":1.0},
}

def sample_next(token, temperature=1.0):
    probs = bigram_probs.get(token, {"<EOS>": 1.0})
    # Apply temperature: logit = log(p), scale, re-softmax
    logits = {k: math.log(v) / temperature for k, v in probs.items()}
    max_l = max(logits.values())
    exp_l = {k: math.exp(v - max_l) for k, v in logits.items()}
    total = sum(exp_l.values())
    normed = {k: v / total for k, v in exp_l.items()}
    r = random.random()
    cumulative = 0.0
    for tok, p in normed.items():
        cumulative += p
        if r <= cumulative:
            return tok
    return list(normed)[-1]

# Autoregressive loop
sequence = ["The"]
for step in range(10):
    last = sequence[-1]
    next_tok = sample_next(last, temperature=1.0)  # Try 0.1 or 2.0
    print(f"Step {step+1}: {' '.join(sequence)} → [{next_tok}]")
    sequence.append(next_tok)
    if next_tok == "<EOS>": break

print(f"\nFinal: {' '.join(sequence)}")
# Try changing temperature=1.0 to 0.1 (greedy) or 2.0 (chaotic) to see diversity change.
